安装
ln -s redis-5.0.5/
yum install gcc
make MALLOC=libc

配置文件启动
mkdir config
cat redis.conf | grep -v "#" | grep -v "^$" > config/redis-6379.conf
cd config
vi redis-6379.conf
	daemonize yes
	port 6379
	dir "/root/redis-5.0.5/data"
	logfile "6379.log"
redis-server config/redis-6379.conf
